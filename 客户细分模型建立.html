from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
from sklearn.metrics import davies_bouldin_score, calinski_harabasz_score, silhouette_score
import matplotlib.pyplot as plt

# 解决中文显示问题
plt.rcParams['font.sans-serif'] = ['SimHei']  
plt.rcParams['axes.unicode_minus'] = False 

# 将数据进行标准化处理
scaler = StandardScaler()
rfm_scaled = scaler.fit_transform(rfm[['Recency', 'Frequency', 'Monetary']])

# 定义要测试的簇数量范围
clusters_range = range(2, 11)

# 初始化三个指标的列表
dbi_values = []
chi_values = []
silhouette_values = []

# 对不同簇数量进行 KMeans 聚类，并计算指标
for k in clusters_range:
    kmeans = KMeans(n_clusters=k, init='k-means++', random_state=0)
    labels = kmeans.fit_predict(rfm_scaled)

    # 计算 DBI, CHI 和 Silhouette Score
    dbi = davies_bouldin_score(rfm_scaled, labels)
    chi = calinski_harabasz_score(rfm_scaled, labels) / 5000  # 调整CHI以便于显示
    silhouette = silhouette_score(rfm_scaled, labels)

    dbi_values.append(dbi)
    chi_values.append(chi)
    silhouette_values.append(silhouette)

# 输出指标值
print("DBI:", dbi_values)
print("CHI:", chi_values)
print("Silhouette Score:", silhouette_values)

# 绘制指标图
plt.figure(figsize=(12, 6))
# 绘制 DBI
plt.plot(clusters_range, dbi_values, marker='o', linestyle='-', label='DBI', color='b')
# 绘制 CHI
plt.plot(clusters_range, chi_values, marker='s', linestyle='-', label='CHI(缩小5000倍)', color='g')
# 绘制 轮廓系数
plt.plot(clusters_range, silhouette_values, marker='^', linestyle='-', label='Silhouette Score', color='r')

# 添加图例和标签
plt.xlabel('聚类数据')
plt.ylabel('得分')
plt.legend()
plt.grid(True)
plt.show()
